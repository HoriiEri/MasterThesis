  \advance\leftskip1cm
  \advance\rightskip1cm
  \vskip\abovecaptionskip
  \sbox\@tempboxa{{\bf #1}\hskip1zw{\gt #2}}%
  \ifdim \wd\@tempboxa >\hsize
    {\bf #1}\hskip1zw{\gt #2}\par

  \fi
  \vskip\belowcaptionskip}
